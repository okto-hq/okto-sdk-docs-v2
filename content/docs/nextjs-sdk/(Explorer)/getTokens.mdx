<Tab value="TypeScript">
```typescript
import { useOkto } from '@okto_web3/react-sdk';
import { getTokens } from '@okto_web3/react-sdk'; // [!code highlight]

function TokensList() {
    const oktoClient = useOkto();

    async function fetchTokens() {
        try {
            const tokens = await getTokens(oktoClient); // [!code highlight]
            console.log('Supported tokens:', tokens);
        } catch (error) {
            console.error('Error fetching tokens:', error);
        }
    }

    return (
        <button onClick={fetchTokens}>
            Fetch Supported Tokens
        </button>
    );
}
```

## Get Tokens

<sub><i>async</i></sub> `getTokens(oktoClient: OktoClient)` retrieves information about all supported tokens.

### Parameters

| Parameter | Type | Description | Required |
|-----------|------|-------------|-----------|
| `oktoClient` | `OktoClient` | Instance of OktoClient obtained from useOkto hook | Yes |

### Response

<Callout title="Success Response">

| Field Name | Type | Description |
|------------|------|-------------|
| `tokens` | [`Promise<Token[]>`](/docs/nextjs-sdk/technical-reference#models) | Array of supported tokens |
